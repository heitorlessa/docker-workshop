Flask hello world

FROM ubuntu:latest
RUN apt-get update -y \
    && apt-get install -y python-pip python-dev build-essential
COPY ./app.py /app
COPY ./requirements.txt /app
WORKDIR /app
RUN pip install -r requirements.txt
ENTRYPOINT ["python"]
CMD ["app.py"]
